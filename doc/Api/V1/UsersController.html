def login
  @user = User.find_by(email: params[:email].downcase)

  if @user && @user.authenticate(params[:password])
    @auth_token = JsonWebToken.encode({user_id: @user.id})
    render status: :ok
  else
    render json: {errors: 'Invalid username / password'}, status: :ok
  end

end
